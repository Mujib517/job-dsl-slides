DSL Example


job('ci') {
    description 'Build and test the app.'
    triggers {
      	scm 'H/5 * * * *'
    }
    scm {
        github 'sheehan/job-dsl-playground'
    }
    steps {
        gradle 'test'
    }
    publishers {
        archiveJunit 'build/test-results/**/*.xml'
        mailer 'mr.sheehan@gmail.com'
    }
}
